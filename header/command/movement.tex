\newcounter{noWall}
\newcommand{\tryWall}[2]{%
    \setcounter{noWall}{1}%
    \foreach \wall in \walls{%
        \listElement{\wall}{0}%
        % \theelement
        \ifnum \theelement = \newX{#1}
            \listElement{\wall}{1}%
            \ifnum \theelement = \newY{#2}
                \setcounter{noWall}{0}%
            \fi
        \fi
    }%
    % \thetrue%
}

\newcommand{\setSpawn}{
    \listElement{\spawn}{0}
    \edef\x{\theelement}
    \listElement{\spawn}{1}
    \edef\y{\theelement}
    % \setcounter{world}{0}
}

\newcounter{lose}
\newcommand{\tryLose}{%
    \setcounter{lose}{0}%
    \tryDeathTrap%
    \setcounter{lose}{\thedeathTrap}%
}

\newcounter{deathTrap}
\newcommand{\tryDeathTrap}{%
    \setcounter{deathTrap}{0}%
    \foreach \spike in \spikes{%
        \listElement{\spike}{0}%
        % \theelement
        \ifnum \theelement = \x
            \listElement{\spike}{1}%
            \ifnum \theelement = \y
                \setcounter{deathTrap}{1}%
            \fi
        \fi
    }%
}

\newcounter{win}
\newcommand{\tryWin}{%
    \setcounter{win}{0}%
    \listElement{\goal}{0}%
    \ifnum \x = \theelement
        \listElement{\goal}{1}%
        \ifnum \y = \theelement
            \setcounter{win}{1}%
        \fi
    \fi
}

\newcommand{\newX}[1]{%
    \ifnum \thetoricWorld = 1
        \loopValue{\ncalc{\x+#1}}{0}{\ncalc{\gridX-1}}%
    \else
        \boundValue{\ncalc{\x+#1}}{0}{\ncalc{\gridX-1}}%
    \fi
}

\newcommand{\newY}[1]{%
    \ifnum \thetoricWorld = 1
        \loopValue{\ncalc{\y+#1}}{0}{\ncalc{\gridY-1}}%
    \else
        \boundValue{\ncalc{\y+#1}}{0}{\ncalc{\gridY-1}}%
    \fi
}

\newcommand{\newWorld}[1]{%
    \loopValue{\ncalc{\theworld+#1}}{0}{\ncalc{\theworldCount-1}}%
}

\newcommand{\move}[3]{%
    \newX{#1};\newY{#2};\newWorld{#3}%
}
